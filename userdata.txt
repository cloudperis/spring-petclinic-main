#!/bin/bash
# install updates
BUCKET=olulegends3
yum update -y

sudo amazon-linux-extras install java-openjdk11

cd ~

rm ~/*

OBJECT="$(aws s3 ls $BUCKET --recursive | sort | tail -n 1 | awk '{print $4}')"
aws s3 cp s3://$BUCKET/$OBJECT . --region=us-east-2
app="$(ls)"

java -jar $app
